170 assert(parent_instruction.isMoveInstruction());
171 const auto& base_instruction = parent_instruction.as<MoveInstructionPoly>();

178 if (mi.tcp_frame.empty())
179 throw std::runtime_error("Descartes, tcp_frame is empty!");

181 if (mi.working_frame.empty())
182 throw std::runtime_error("Descartes, working_frame is empty!");
183
184 Eigen::Isometry3d tcp_offset = prob.env->findTCPOffset(mi);
185 std::vector<std::string> joint_names = prob.manip->getJointNames();
186
187 /* Check if this cartesian waypoint is dynamic

189 // auto it = std::find(active_links.begin(), active_links.end(), prob.manip_inv_kin->getBaseLinkName());

191 // throw std::runtime_error("DescartesDefaultPlanProfile: Assigned dynamic waypoint but parent instruction working
192 // is "
193 // "not set to the base link of manipulator!");
194
195 DescartesCollision::Ptr ci = nullptr;
196 if (enable_collision)
197 ci = std::make_shared<DescartesCollision>(*prob.env, prob.manip, vertex_collision_check_config, debug);
198
199 // Add vertex evaluator
200 std::shared_ptr<descartes_light::WaypointSampler<FloatType>> sampler;
201 if (vertex_evaluator == nullptr)

203 auto ve = std::make_shared<DescartesJointLimitsVertexEvaluator>(prob.manip->getLimits().joint_limits);
204 sampler = std::make_shared<DescartesRobotSampler<FloatType>>(mi.working_frame,
