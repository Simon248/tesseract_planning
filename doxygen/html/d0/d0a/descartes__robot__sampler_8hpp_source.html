69template <typename FloatType>
70std::vector<descartes_light::StateSample<FloatType>> DescartesRobotSampler<FloatType>::sample() const
71{
72 // Generate all possible Cartesian poses
73 tesseract_common::VectorIsometry3d target_poses = target_pose_sampler_(target_pose_);
74
75 // Generate the IK solutions for those poses
76 std::vector<descartes_light::StateSample<FloatType>> samples;
77 for (const auto& pose : target_poses)
78 {
79 // Get the transformation to the kinematic tip link
80 Eigen::Isometry3d target_pose = pose * tcp_offset_.inverse();
81
82 // Solve IK (TODO Should tcp_offset be stored in KinGroupIKInput?)
83 tesseract_kinematics::KinGroupIKInput ik_input(target_pose, target_working_frame_, tcp_frame_);
84 tesseract_kinematics::IKSolutions ik_solutions = manip_->calcInvKin({ ik_input }, ik_seed_);
85
86 if (ik_solutions.empty())
87 continue;
88
89 // Check each individual joint solution
90 for (const auto& sol : ik_solutions)
91 {
92 if ((is_valid_ != nullptr) && !(*is_valid_)(sol))
93 continue;
94
95 auto state = std::make_shared<descartes_light::State<FloatType>>(sol.cast<FloatType>());
96 if (allow_collision_ && collision_ == nullptr)
97 {
98 samples.push_back(descartes_light::StateSample<FloatType>{ state, static_cast<FloatType>(0.0) });
99 }
100 else if (!allow_collision_)
101 {
102 if (collision_->validate(sol))
103 samples.push_back(descartes_light::StateSample<FloatType>{ state, 0.0 });
104 }
105 else
106 {
107 const FloatType cost = static_cast<FloatType>(collision_->distance(sol));
108 samples.push_back(descartes_light::StateSample<FloatType>{ state, cost });
109 }
110 }
111 }
112
113 if (samples.empty())
114 return samples;
115
116 if (allow_collision_)
117 {
118 // Sort state samples in descending order by distance from nearest collision (i.e. state cost)
119 std::sort(samples.begin(), samples.end(), [](const auto& a, const auto& b) { return a.cost > b.cost; });
120
121 // Convert the distance into a cost and normalize
122 if (samples.size() > 1)
123 {
124 const FloatType max_dist = samples.front().cost;
125 const FloatType min_dist = samples.back().cost;
126 const FloatType range = max_dist - min_dist;
127 if (range > std::numeric_limits<FloatType>::epsilon())
128 {
129 std::for_each(samples.begin(), samples.end(), [&min_dist, &range](auto& sample) {
130 sample.cost = static_cast<FloatType>(1.0) - (sample.cost - min_dist) / range;
131 });
132 }
133 else
134 {
135 std::for_each(samples.begin(), samples.end(), [](auto& sample) { sample.cost = 0.0; });
136 }
137 }
138 }
139
140 // Generate the redundant solutions
141 if (use_redundant_joint_solutions_)
142 {
143 const Eigen::MatrixX2d& limits = manip_->getLimits().joint_limits;
144 std::vector<Eigen::Index> redundancy_capable_joints = manip_->getRedundancyCapableJointIndices();
145 std::vector<descartes_light::StateSample<FloatType>> redundant_samples;
146 for (const descartes_light::StateSample<FloatType>& sample : samples)
147 {
148 const auto redundant_solutions =
149 tesseract_kinematics::getRedundantSolutions<FloatType>(*(sample.state), limits, redundancy_capable_joints);
150
151 // Add the redundant samples with the same cost as the nominal sample
152 std::transform(redundant_solutions.begin(),
153 redundant_solutions.end(),
154 std::back_inserter(redundant_samples),
155 [&sample](const auto& sol) {
156 auto state = std::make_shared<descartes_light::State<FloatType>>(sol.template cast<FloatType>());
157 return descartes_light::StateSample<FloatType>{ state, sample.cost };
158 });
159 }
160
161 // Combine the nominal samples with the redundant samples
162 samples.insert(samples.end(), redundant_samples.begin(), redundant_samples.end());
163 }
164
165 return samples;
166}
