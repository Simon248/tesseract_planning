62template <typename FloatType>
63PlannerResponse DescartesMotionPlanner<FloatType>::solve(const PlannerRequest& request) const
64{
65 PlannerResponse response;
66 std::shared_ptr<DescartesProblem<FloatType>> problem;
67 if (request.data)
68 {
69 problem = std::static_pointer_cast<DescartesProblem<FloatType>>(request.data);
70 }
71 else
72 {
73 try
74 {
75 problem = createProblem(request);
76 }
77 catch (std::exception& e)
78 {
79 CONSOLE_BRIDGE_logError("DescartesMotionPlanner failed to generate problem: %s.", e.what());
80 response.successful = false;
81 response.message = ERROR_INVALID_INPUT;
82 return response;
83 }
84
85 response.data = problem;
86 }
87
88 descartes_light::SearchResult<FloatType> descartes_result;
89 try
90 {
91 descartes_light::LadderGraphSolver<FloatType> solver(problem->num_threads);
92 solver.build(problem->samplers, problem->edge_evaluators, problem->state_evaluators);
93 descartes_result = solver.search();
94 if (descartes_result.trajectory.empty())
95 {
96 CONSOLE_BRIDGE_logError("Search for graph completion failed");
97 response.successful = false;
98 response.message = ERROR_FAILED_TO_FIND_VALID_SOLUTION;
99 return response;
100 }
101 }
102 catch (...)
103 {
104 // CONSOLE_BRIDGE_logError("Failed to build vertices");
105 // for (const auto& i : graph_builder.getFailedVertices())
106 // response.failed_waypoints.push_back(config_->waypoints[i]);
107
108 // // Copy the waypoint if it is not already in the failed waypoints list
109 // std::copy_if(config_->waypoints.begin(),
110 // config_->waypoints.end(),
111 // std::back_inserter(response.succeeded_waypoints),
112 // [&response](const Waypoint::ConstPtr wp) {
113 // return std::find(response.failed_waypoints.begin(), response.failed_waypoints.end(), wp) ==
114 // response.failed_waypoints.end();
115 // });
116
117 response.successful = false;
118 response.message = ERROR_FAILED_TO_BUILD_GRAPH;
119 return response;
120 }
121
122 const std::vector<std::string> joint_names = problem->manip->getJointNames();
123 const Eigen::MatrixX2d joint_limits = problem->manip->getLimits().joint_limits;
124
125 // Enforce limits
126 std::vector<Eigen::VectorXd> solution{};
127 solution.reserve(descartes_result.trajectory.size());
128 for (const auto& js : descartes_result.trajectory)
129 {
130 solution.push_back(js->values.template cast<double>());
131 assert(tesseract_common::satisfiesPositionLimits<double>(solution.back(), joint_limits));
132 tesseract_common::enforcePositionLimits<double>(solution.back(), joint_limits);
133 }
134
135 // Flatten the results to make them easier to process
136 response.results = request.instructions;
137 auto results_instructions = response.results.flatten(&moveFilter);
138
139 // Loop over the flattened results and add them to response if the input was a plan instruction
140 std::size_t result_index{ 0 };
141 for (std::size_t idx = 0; idx < results_instructions.size(); idx++)
142 {
143 // If idx is zero then this should be the start instruction
144 assert((idx == 0) ? results_instructions.at(idx).get().isMoveInstruction() : true);
145 if (results_instructions.at(idx).get().isMoveInstruction())
146 {
147 auto& move_instruction = results_instructions.at(idx).get().as<MoveInstructionPoly>();
148 if (move_instruction.getWaypoint().isCartesianWaypoint())
149 {
150 assignSolution(move_instruction, joint_names, solution[result_index++], request.format_result_as_input);
151 }
152 else if (move_instruction.getWaypoint().isJointWaypoint())
153 {
154 auto& jwp = move_instruction.getWaypoint().as<JointWaypointPoly>();
155 if (jwp.isConstrained())
156 {
157 assignSolution(move_instruction, joint_names, solution[result_index++], request.format_result_as_input);
158 continue;
159 }
160
161 const Eigen::VectorXd& start_state = solution[result_index - 1];
162 Eigen::Index cnt = 1;
163 bool is_constrained = jwp.isConstrained();
164 while (!is_constrained)
165 {
166 auto temp = results_instructions.at(idx + static_cast<std::size_t>(cnt++)).get().as<MoveInstructionPoly>();
167 if (temp.getWaypoint().isCartesianWaypoint() || temp.getWaypoint().isStateWaypoint())
168 is_constrained = true;
169 else if (temp.getWaypoint().isJointWaypoint())
170 is_constrained = temp.getWaypoint().as<JointWaypointPoly>().isConstrained();
171 else
172 throw std::runtime_error("Unsupported Waypoint Type!");
173 }
174 const Eigen::VectorXd& end_state = solution[result_index];
175
176 Eigen::MatrixXd states = interpolate(start_state, end_state, cnt);
177 for (Eigen::Index i = 0; i < cnt - 1; ++i)
178 {
179 if (i != 0)
180 ++idx;
181 auto& interp_mi = results_instructions.at(idx).get().as<MoveInstructionPoly>();
182 assignSolution(interp_mi, joint_names, states.col(i + 1), request.format_result_as_input);
183 }
184 }
185 else if (move_instruction.getWaypoint().isStateWaypoint())
186 {
187 ++result_index;
188 }
189 else
190 {
191 throw std::runtime_error("Unsupported Waypoint Type!");
192 }
193 }
194 }
195
196 response.successful = true;
197 response.message = SOLUTION_FOUND;
198 return response;
199}

219template <typename FloatType>
220std::shared_ptr<DescartesProblem<FloatType>>
221DescartesMotionPlanner<FloatType>::createProblem(const PlannerRequest& request) const
222{
223 auto prob = std::make_shared<DescartesProblem<FloatType>>();
224
225 // Clear descartes data
226 prob->edge_evaluators.clear();
227 prob->samplers.clear();
228 prob->state_evaluators.clear();
229
230 // Assume all the plan instructions have the same manipulator as the composite
231 assert(!request.instructions.getManipulatorInfo().empty());
232 const tesseract_common::ManipulatorInfo& composite_mi = request.instructions.getManipulatorInfo();
233
234 if (composite_mi.manipulator.empty())
235 throw std::runtime_error("Descartes, manipulator is empty!");
236
237 // Get Manipulator Information
238 try
239 {
240 if (composite_mi.manipulator_ik_solver.empty())
241 prob->manip = request.env->getKinematicGroup(composite_mi.manipulator);
242 else
243 prob->manip = request.env->getKinematicGroup(composite_mi.manipulator, composite_mi.manipulator_ik_solver);
244 }
245 catch (...)
246 {
247 throw std::runtime_error("Descartes problem generator failed to create kinematic group!");
248 }
249
250 if (!prob->manip)
251 {
252 CONSOLE_BRIDGE_logError("No Kinematics Group found");
253 return prob;
254 }
255
256 prob->env_state = request.env_state;
257 prob->env = request.env;
258
259 std::vector<std::string> joint_names = prob->manip->getJointNames();
260
261 // Flatten the input for planning
262 auto move_instructions = request.instructions.flatten(&moveFilter);
263
264 // Transform plan instructions into descartes samplers
265 int index = 0;
266 for (const auto& move_instruction : move_instructions)
267 {
268 const auto& instruction = move_instruction.get();
269
270 assert(instruction.isMoveInstruction());
271 const auto& plan_instruction = instruction.template as<MoveInstructionPoly>();
272
273 // If plan instruction has manipulator information then use it over the one provided by the composite.
274 tesseract_common::ManipulatorInfo mi = composite_mi.getCombined(plan_instruction.getManipulatorInfo());
275
276 if (mi.manipulator.empty())
277 throw std::runtime_error("Descartes, manipulator is empty!");
278
279 if (mi.tcp_frame.empty())
280 throw std::runtime_error("Descartes, tcp_frame is empty!");
281
282 if (mi.working_frame.empty())
283 throw std::runtime_error("Descartes, working_frame is empty!");
284
285 // Get Plan Profile
286 std::string profile = plan_instruction.getProfile();
287 profile = getProfileString(name_, profile, request.plan_profile_remapping);
288 auto cur_plan_profile = getProfile<DescartesPlanProfile<FloatType>>(
289 name_, profile, *request.profiles, std::make_shared<DescartesDefaultPlanProfile<FloatType>>());
290 // cur_plan_profile = applyProfileOverrides(name_, profile, cur_plan_profile,
291 // plan_instruction.profile_overrides);
292 if (!cur_plan_profile)
293 throw std::runtime_error("DescartesMotionPlanner: Invalid profile");
294
295 if (plan_instruction.getWaypoint().isCartesianWaypoint())
296 {
297 const auto& cur_wp = plan_instruction.getWaypoint().template as<CartesianWaypointPoly>();
298 cur_plan_profile->apply(*prob, cur_wp.getTransform(), plan_instruction, composite_mi, index);
299 ++index;
300 }
301 else if (plan_instruction.getWaypoint().isJointWaypoint())
302 {
303 if (plan_instruction.getWaypoint().as<JointWaypointPoly>().isConstrained())
304 {
305 assert(checkJointPositionFormat(joint_names, plan_instruction.getWaypoint()));
306 const Eigen::VectorXd& cur_position = getJointPosition(plan_instruction.getWaypoint());
307 cur_plan_profile->apply(*prob, cur_position, plan_instruction, composite_mi, index);
308 ++index;
309 }
310 }
311 else if (plan_instruction.getWaypoint().isStateWaypoint())
312 {
313 assert(checkJointPositionFormat(joint_names, plan_instruction.getWaypoint()));
314 const Eigen::VectorXd& cur_position = getJointPosition(plan_instruction.getWaypoint());
315 cur_plan_profile->apply(*prob, cur_position, plan_instruction, composite_mi, index);
316 ++index;
317 }
318 else
319 {
320 throw std::runtime_error("DescartesMotionPlanner: unknown waypoint type");
321 }
322 }
323
324 // Call the base class generate which checks the problem to make sure everything is in order
325 return prob;
326}
