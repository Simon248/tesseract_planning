66template <typename FloatType>
67tesseract_common::StatusCode DescartesMotionPlanner<FloatType>::solve(const PlannerRequest& request,
68 PlannerResponse& response,
69 const bool /*verbose*/) const
70{
71 std::shared_ptr<DescartesProblem<FloatType>> problem;
72 if (request.data)
73 {
74 problem = std::static_pointer_cast<DescartesProblem<FloatType>>(request.data);
75 }
76 else
77 {
78 try
79 {
80 problem = createProblem(request);
81 }
82 catch (std::exception& e)
83 {
84 CONSOLE_BRIDGE_logError("DescartesMotionPlanner failed to generate problem: %s.", e.what());
85 response.status =
86 tesseract_common::StatusCode(DescartesMotionPlannerStatusCategory::ErrorInvalidInput, status_category_);
87 return response.status;
88 }
89
90 response.data = problem;
91 }
92
93 descartes_light::SearchResult<FloatType> descartes_result;
94 try
95 {
96 descartes_light::LadderGraphSolver<FloatType> solver(problem->num_threads);
97 solver.build(problem->samplers, problem->edge_evaluators, problem->state_evaluators);
98 descartes_result = solver.search();
99 if (descartes_result.trajectory.empty())
100 {
101 CONSOLE_BRIDGE_logError("Search for graph completion failed");
102 response.status = tesseract_common::StatusCode(
103 DescartesMotionPlannerStatusCategory::ErrorFailedToFindValidSolution, status_category_);
104 return response.status;
105 }
106 }
107 catch (...)
108 {
109 // CONSOLE_BRIDGE_logError("Failed to build vertices");
110 // for (const auto& i : graph_builder.getFailedVertices())
111 // response.failed_waypoints.push_back(config_->waypoints[i]);
112
113 // // Copy the waypoint if it is not already in the failed waypoints list
114 // std::copy_if(config_->waypoints.begin(),
115 // config_->waypoints.end(),
116 // std::back_inserter(response.succeeded_waypoints),
117 // [&response](const Waypoint::ConstPtr wp) {
118 // return std::find(response.failed_waypoints.begin(), response.failed_waypoints.end(), wp) ==
119 // response.failed_waypoints.end();
120 // });
121
122 response.status =
123 tesseract_common::StatusCode(DescartesMotionPlannerStatusCategory::ErrorFailedToBuildGraph, status_category_);
124 return response.status;
125 }
126
127 // Enforce limits
128 std::vector<Eigen::VectorXd> solution{};
129 solution.reserve(descartes_result.trajectory.size());
130 for (const auto& js : descartes_result.trajectory)
131 {
132 solution.push_back(js->values.template cast<double>());
133 // Using 1e-6 because when using floats with descartes epsilon does not seem to be enough
134 assert(
135 tesseract_common::satisfiesPositionLimits<double>(solution.back(), problem->manip->getLimits().joint_limits));
136 tesseract_common::enforcePositionLimits<double>(solution.back(), problem->manip->getLimits().joint_limits);
137 }
138
139 // Flatten the results to make them easier to process
140 response.results = request.seed;
141 auto results_flattened = flattenProgramToPattern(response.results, request.instructions);
142 auto instructions_flattened = flattenProgram(request.instructions);
143
144 // Loop over the flattened results and add them to response if the input was a plan instruction
145 std::size_t result_index = 0;
146 for (std::size_t idx = 0; idx < instructions_flattened.size(); idx++)
147 {
148 // If idx is zero then this should be the start instruction
149 assert((idx == 0) ? isMoveInstruction(instructions_flattened.at(idx).get()) : true);
150 assert((idx == 0) ? isMoveInstruction(results_flattened[idx].get()) : true);
151 if (isMoveInstruction(instructions_flattened.at(idx).get()))
152 {
153 const auto& plan_instruction = instructions_flattened.at(idx).get().as<MoveInstruction>();
154 if (plan_instruction.isStart())
155 {
156 assert(idx == 0);
157 assert(isMoveInstruction(results_flattened[idx].get()));
158 auto& move_instruction = results_flattened[idx].get().as<MoveInstruction>();
159
160 auto& swp = move_instruction.getWaypoint().as<StateWaypoint>();
161 swp.position = solution[result_index++];
162 assert(swp.joint_names == problem->manip->getJointNames());
163 }
164 else if (plan_instruction.isLinear())
165 {
166 // This instruction corresponds to a composite. Set all results in that composite to the results
167 assert(isCompositeInstruction(results_flattened[idx].get()));
168 auto& move_instructions = results_flattened[idx].get().as<CompositeInstruction>();
169 for (auto& instruction : move_instructions)
170 {
171 auto& swp = instruction.as<MoveInstruction>().getWaypoint().as<StateWaypoint>();
172 swp.position = solution[result_index++];
173 assert(swp.joint_names == problem->manip->getJointNames());
174 }
175 }
176 else if (plan_instruction.isFreespace())
177 {
178 assert(result_index > 0);
179 // Because descartes does not support freespace it just includes the plan instruction waypoint so we will
180 // fill out the results with a joint interpolated trajectory.
181 Eigen::VectorXd& start = solution[result_index - 1];
182 Eigen::VectorXd& stop = solution[result_index++];
183
184 // This instruction corresponds to a composite. Set all results in that composite to the results
185 assert(isCompositeInstruction(results_flattened[idx].get()));
186 auto& move_instructions = results_flattened[idx].get().as<CompositeInstruction>();
187
188 Eigen::MatrixXd temp = interpolate(start, stop, static_cast<int>(move_instructions.size()));
189
190 assert(temp.cols() == static_cast<long>(move_instructions.size()) + 1);
191 for (std::size_t i = 0; i < move_instructions.size(); ++i)
192 {
193 auto& swp = move_instructions[i].as<MoveInstruction>().getWaypoint().as<StateWaypoint>();
194 swp.position = temp.col(static_cast<long>(i) + 1);
195 assert(swp.joint_names == problem->manip->getJointNames());
196 }
197 }
198 else
199 {
200 throw std::runtime_error("Unsupported Plan Instruction Type!");
201 }
202 }
203 }
204
205 response.status = tesseract_common::StatusCode(DescartesMotionPlannerStatusCategory::SolutionFound, status_category_);
206 return response.status;
