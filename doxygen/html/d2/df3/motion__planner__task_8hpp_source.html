83 TaskComposerNodeInfo::UPtr runImpl(TaskComposerInput& input,

85 {
86 auto info = std::make_unique<TaskComposerNodeInfo>(*this);
87 info->return_value = 0;
88 info->env = input.problem.env;
89
90 if (input.isAborted())
91 {
92 info->message = "Aborted";
93 return info;
94 }

96 tesseract_common::Timer timer;
97 timer.start();
98
99 // --------------------
100 // Check that inputs are valid
101 // --------------------
102 auto input_data_poly = input.data_storage.getData(input_keys_[0]);
103 if (input_data_poly.isNull() || input_data_poly.getType() != std::type_index(typeid(CompositeInstruction)))
104 {
105 info->message = "Input instructions to MotionPlannerTask: " + name_ + " must be a composite instruction";
106 info->elapsed_time = timer.elapsedSeconds();
107 CONSOLE_BRIDGE_logError("%s", info->message.c_str());
108 return info;
109 }
110
111 // Make a non-const copy of the input instructions to update the start/end
112 auto& instructions = input_data_poly.template as<CompositeInstruction>();
113 assert(!(input.problem.manip_info.empty() && instructions.getManipulatorInfo().empty()));
114 instructions.setManipulatorInfo(instructions.getManipulatorInfo().getCombined(input.problem.manip_info));
115
116 // --------------------
117 // Fill out request
118 // --------------------
119 PlannerRequest request;
120 request.env_state = input.problem.env->getState();
121 request.env = input.problem.env;
122 request.instructions = instructions;
123 request.profiles = input.profiles;
124 request.plan_profile_remapping = input.problem.move_profile_remapping;
125 request.composite_profile_remapping = input.problem.composite_profile_remapping;
126 request.format_result_as_input = format_result_as_input_;
127
128 // --------------------
129 // Fill out response
130 // --------------------
131 request.verbose = false;
132 if (console_bridge::getLogLevel() == console_bridge::LogLevel::CONSOLE_BRIDGE_LOG_DEBUG)
133 request.verbose = true;
134 PlannerResponse response = planner_->solve(request);
135
136 // --------------------
137 // Verify Success
138 // --------------------
139 if (response)
140 {
141 input.data_storage.setData(output_keys_[0], response.results);
142
143 info->return_value = 1;
144 info->message = response.message;
145 info->elapsed_time = timer.elapsedSeconds();
146 CONSOLE_BRIDGE_logDebug("Motion Planner process succeeded");
147 return info;
148 }
149
150 CONSOLE_BRIDGE_logInform("%s motion planning failed (%s) for process input: %s",
151 planner_->getName().c_str(),
152 response.message.c_str(),
153 instructions.getDescription().c_str());
154 info->message = response.message;
155 info->elapsed_time = timer.elapsedSeconds();
156 return info;
157 }
