73template <typename FloatType>
74inline RobotConfig getRobotConfig(const tesseract_kinematics::JointGroup& joint_group,
75 const std::string& base_link,
76 const std::string& tcp_frame,
77 const Eigen::Ref<const Eigen::Matrix<FloatType, Eigen::Dynamic, 1>>& joint_values,
78 const Eigen::Ref<const Eigen::Vector2i>& sign_correction = Eigen::Vector2i::Ones())
79{
80 // Get state
81 tesseract_common::TransformMap state = joint_group.calcFwdKin(joint_values.template cast<double>());
82
83 // Get the pose at tool0
84 Eigen::Isometry3d pose = state.at(base_link).inverse() * state.at(tcp_frame);
85
86 // Get the base rotated by joint 1
87 Eigen::Isometry3d prime_pose(
88 Eigen::AngleAxisd(static_cast<double>(joint_values.tail(6)(0)), Eigen::Vector3d::UnitZ()));
89
90 // Transform tool0 pose into new frame
91 Eigen::Isometry3d pose_prime = prime_pose.inverse() * pose;
92
93 // If pose_prime.x is greater than and equal to zero then it is in the forward configuration, otherwise
94 // in the backward configuration.
95
96 std::array<std::string, 3> config;
97
98 // Wrist Flip or Not Flip
99 if ((sign_correction[1] * joint_values(4)) >= 0)
100 config[0] = "F";
101 else
102 config[0] = "N";
103
104 // Elbow Up or Down
105 if ((sign_correction[0] * joint_values(2)) < M_PI / 2)
106 config[1] = "U";
107 else
108 config[1] = "D";
109
110 // Robot Front or Back
111 if (pose_prime.translation().x() >= 0)
112 config[2] = "T";
113 else
114 config[2] = "B";
115
116 if (config == std::array<std::string, 3>({ "F", "U", "T" }))
117 return RobotConfig::FUT;
118
119 if (config == std::array<std::string, 3>({ "N", "U", "T" }))
120 return RobotConfig::NUT;
121
122 if (config == std::array<std::string, 3>({ "F", "D", "T" }))
123 return RobotConfig::FDT;
124
125 if (config == std::array<std::string, 3>({ "N", "D", "T" }))
126 return RobotConfig::NDT;
127
128 if (config == std::array<std::string, 3>({ "F", "U", "B" }))
129 return RobotConfig::FUB;
130
131 if (config == std::array<std::string, 3>({ "N", "U", "B" }))
132 return RobotConfig::NUB;
133
134 if (config == std::array<std::string, 3>({ "F", "D", "B" }))
135 return RobotConfig::FDB;
136
137 return RobotConfig::NDB;
138}
