62template <typename FloatType>
63std::pair<bool, FloatType>
64DescartesCollisionEdgeEvaluator<FloatType>::evaluate(const descartes_light::State<FloatType>& start,
65 const descartes_light::State<FloatType>& end) const
66{
67 assert(start.values.rows() == end.values.rows());
68
69 // Happens in two phases:
70 // 1. Compute the transform of all objects
71 tesseract_common::TrajArray segment(2, start.values.rows());
72 for (Eigen::Index i = 0; i < start.values.rows(); ++i)
73 {
74 segment(0, i) = start[i];
75 segment(1, i) = end[i];
76 }
77
78 std::vector<tesseract_collision::ContactResultMap> discrete_results;
79 std::vector<tesseract_collision::ContactResultMap> continuous_results;
80 bool discrete_in_contact = discreteCollisionCheck(discrete_results, segment, allow_collision_);
81 bool continuous_in_contact = continuousCollisionCheck(continuous_results, segment, allow_collision_);
82
83 if (!discrete_in_contact && !continuous_in_contact)
84 return std::make_pair(true, 0);
85
86 // TODO: Update this to consider link pairs
87 auto collision_safety_margin_ =
88 static_cast<FloatType>(collision_check_config_.contact_manager_config.margin_data.getMaxCollisionMargin());
89
90 if (!discrete_in_contact && continuous_in_contact && allow_collision_)
91 return std::make_pair(true, collision_safety_margin_ - continuous_results.begin()->begin()->second[0].distance);
92
93 if (discrete_in_contact && !continuous_in_contact && allow_collision_)
94 return std::make_pair(true, collision_safety_margin_ - discrete_results.begin()->begin()->second[0].distance);
95
96 if (discrete_in_contact && continuous_in_contact && allow_collision_)
97 {
98 double d = collision_safety_margin_ - discrete_results.begin()->begin()->second[0].distance;
99 double c = collision_safety_margin_ - continuous_results.begin()->begin()->second[0].distance;
100 return std::make_pair(true, std::max(d, c));
101 }
102
103 return std::make_pair(false, 0);
104}

106template <typename FloatType>
107bool DescartesCollisionEdgeEvaluator<FloatType>::continuousCollisionCheck(
108 std::vector<tesseract_collision::ContactResultMap>& results,
109 const tesseract_common::TrajArray& segment,
110 bool find_best) const
111{
112 // It was time using chronos time elapsed and it was faster to cache the contact manager
113 unsigned long int hash = std::hash<std::thread::id>{}(std::this_thread::get_id());
114 tesseract_collision::ContinuousContactManager::Ptr cm;
115 mutex_.lock();
116 auto it = continuous_contact_managers_.find(hash);
117 if (it == continuous_contact_managers_.end())
118 {
119 cm = continuous_contact_manager_->clone();
120 continuous_contact_managers_[hash] = cm;
121 }
122 else
123 {
124 cm = it->second;
125 }
126 mutex_.unlock();
127 tesseract_collision::CollisionCheckConfig config = collision_check_config_;
128 if (config.type == tesseract_collision::CollisionEvaluatorType::LVS_DISCRETE ||
129 config.type == tesseract_collision::CollisionEvaluatorType::LVS_CONTINUOUS)
130 config.type = tesseract_collision::CollisionEvaluatorType::LVS_CONTINUOUS;
131 else
132 config.type = tesseract_collision::CollisionEvaluatorType::CONTINUOUS;
133 config.contact_request.type =
134 (find_best) ? tesseract_collision::ContactTestType::CLOSEST : tesseract_collision::ContactTestType::FIRST;
135
136 return tesseract_environment::checkTrajectory(results, *cm, *manip_, segment, config);
137}
