103 {
104 auto info = std::make_unique<TaskComposerNodeInfo>(*this);
105 info->return_value = 0;
106 info->env = input.problem.env;
107
108 if (input.isAborted())
109 {
110 info->message = "Aborted";
111 return info;
112 }
113
114 tesseract_common::Timer timer;
115 timer.start();
116
117 // --------------------
118 // Check that inputs are valid
119 // --------------------
120 auto input_data_poly = input.data_storage.getData(input_keys_[0]);
121 try
122 {
123 checkTaskInput(input_data_poly);
124 }
125 catch (const std::exception& e)
126 {
127 info->message = e.what();
128 info->elapsed_time = timer.elapsedSeconds();
129 CONSOLE_BRIDGE_logError("%s", info->message.c_str());
130 return info;
131 }
132
133 auto& program = input_data_poly.template as<CompositeInstruction>();
134 TaskComposerGraph task_graph;
135
136 tesseract_common::ManipulatorInfo program_manip_info =
137 program.getManipulatorInfo().getCombined(input.problem.manip_info);
138
139 auto start_task = std::make_unique<StartTask>();
140 auto start_uuid = task_graph.addNode(std::move(start_task));
141
142 std::vector<std::pair<boost::uuids::uuid, std::string>> raster_tasks;
143 raster_tasks.reserve(program.size());
144
145 // Generate all of the raster tasks. They don't depend on anything
146 std::size_t raster_idx = 0;
147 for (std::size_t idx = 0; idx < program.size(); idx += 2)
148 {
149 // Get Raster program
150 auto raster_input = program[idx].template as<CompositeInstruction>();
151
152 // Set the manipulator info
153 raster_input.setManipulatorInfo(raster_input.getManipulatorInfo().getCombined(program_manip_info));
154
155 // Insert Start Plan Instruction
156 if (idx > 0)
157 {
158 const InstructionPoly& pre_input_instruction = program[idx - 1];
159 assert(pre_input_instruction.isCompositeInstruction());
160 const auto& tci = pre_input_instruction.as<CompositeInstruction>();
161 const auto* li = tci.getLastMoveInstruction();
162 assert(li != nullptr);
163 raster_input.insertMoveInstruction(raster_input.begin(), *li);
164 }
165
166 auto raster_pipeline_task = std::make_unique<RasterTaskType>("Raster #" + std::to_string(raster_idx + 1) + ": " +
167 raster_input.getDescription());
168 std::string raster_pipeline_key = raster_pipeline_task->getUUIDString();
169 auto raster_pipeline_uuid = task_graph.addNode(std::move(raster_pipeline_task));
170 raster_tasks.emplace_back(raster_pipeline_uuid, raster_pipeline_key);
171 input.data_storage.setData(raster_pipeline_key, raster_input);
172
173 task_graph.addEdges(start_uuid, { raster_pipeline_uuid });
174
175 raster_idx++;
176 }
177
178 // Loop over all transitions
179 std::vector<std::string> transition_keys;
180 transition_keys.reserve(program.size());
181 std::size_t transition_idx = 0;
182 for (std::size_t idx = 1; idx < program.size() - 1; idx += 2)
183 {
184 // Get transition program
185 auto transition_input = program[idx].template as<CompositeInstruction>();
186
187 // Set the manipulator info
188 transition_input.setManipulatorInfo(transition_input.getManipulatorInfo().getCombined(program_manip_info));
189
190 const InstructionPoly& pre_input_instruction = program[idx - 1];
191 assert(pre_input_instruction.isCompositeInstruction());
192 const auto& tci = pre_input_instruction.as<CompositeInstruction>();
193 const auto* li = tci.getLastMoveInstruction();
194 assert(li != nullptr);
195 transition_input.insertMoveInstruction(transition_input.begin(), *li);
196
197 auto transition_pipeline_task = std::make_unique<TransitionTaskType>(
198 "Transition #" + std::to_string(transition_idx + 1) + ": " + transition_input.getDescription());
199 std::string transition_pipeline_key = transition_pipeline_task->getUUIDString();
200 auto transition_pipeline_uuid = task_graph.addNode(std::move(transition_pipeline_task));
201 transition_keys.push_back(transition_pipeline_key);
202
203 const auto& prev = raster_tasks[transition_idx];
204 const auto& next = raster_tasks[transition_idx + 1];
205 auto transition_mux_task =
206 std::make_unique<UpdateStartAndEndStateTask>(prev.second, next.second, transition_pipeline_key, false);
207 std::string transition_mux_key = transition_mux_task->getUUIDString();
208 auto transition_mux_uuid = task_graph.addNode(std::move(transition_mux_task));
209
210 input.data_storage.setData(transition_mux_key, transition_input);
211
212 task_graph.addEdges(transition_mux_uuid, { transition_pipeline_uuid });
213 task_graph.addEdges(prev.first, { transition_mux_uuid });
214 task_graph.addEdges(next.first, { transition_mux_uuid });
215
216 transition_idx++;
217 }
218
219 // Debug remove
220 std::ofstream tc_out_data;
221 tc_out_data.open(tesseract_common::getTempPath() + "task_composer_raster_subgraph_example.dot");
222 task_graph.dump(tc_out_data); // dump the graph including dynamic tasks
223 tc_out_data.close();
224
225 TaskComposerFuture::UPtr future = executor.value().get().run(task_graph, input);
226 future->wait();
227
228 if (input.isAborted())
229 {
230 info->message = "Raster only subgraph failed";
231 info->elapsed_time = timer.elapsedSeconds();
232 CONSOLE_BRIDGE_logError("%s", info->message.c_str());
233 return info;
234 }
235
236 program.clear();
237 for (std::size_t i = 0; i < raster_tasks.size(); ++i)
238 {
239 CompositeInstruction segment = input.data_storage.getData(raster_tasks[i].second).as<CompositeInstruction>();
240 if (i != 0)
241 segment.erase(segment.begin());
242
243 program.emplace_back(segment);
244
245 if (i < raster_tasks.size() - 1)
246 {
247 CompositeInstruction transition = input.data_storage.getData(transition_keys[i]).as<CompositeInstruction>();
248 transition.erase(transition.begin());
249 program.emplace_back(transition);
250 }
251 }
252
253 input.data_storage.setData(output_keys_[0], program);
254
255 info->message = "Successful";
256 info->return_value = 1;
257 info->elapsed_time = timer.elapsedSeconds();
258 return info;
259 }

261 static void checkTaskInput(const tesseract_common::AnyPoly& input)
262 {
263 // -------------
264 // Check Input
265 // -------------
266 if (input.isNull())
267 throw std::runtime_error("RasterOnlyMotionTask, input is null");
268
269 if (input.getType() != std::type_index(typeid(CompositeInstruction)))
270 throw std::runtime_error("RasterOnlyMotionTask, input is not a composite instruction");
271
272 const auto& composite = input.as<CompositeInstruction>();
273
274 // Check rasters and transitions
275 for (const auto& i : composite)
276 {
277 // Both rasters and transitions should be a composite
278 if (!i.isCompositeInstruction())
279 throw std::runtime_error("RasterOnlyMotionTask, Both rasters and transitions should be a composite");
280 }
281 }
