103 TaskComposerNodeInfo::UPtr runImpl(TaskComposerInput& input,
104 OptionalTaskComposerExecutor executor) const override final
105 {
106 auto info = std::make_unique<TaskComposerNodeInfo>(*this);
107 info->return_value = 0;
108 info->env = input.problem.env;
109
110 if (input.isAborted())
111 {
112 info->message = "Aborted";
113 return info;
114 }
115
116 tesseract_common::Timer timer;
117 timer.start();
118
119 // --------------------
120 // Check that inputs are valid
121 // --------------------
122 auto input_data_poly = input.data_storage.getData(input_keys_[0]);
123 try
124 {
125 checkTaskInput(input_data_poly);
126 }
127 catch (const std::exception& e)
128 {
129 info->message = e.what();
130 info->elapsed_time = timer.elapsedSeconds();
131 CONSOLE_BRIDGE_logError("%s", info->message.c_str());
132 return info;
133 }
134
135 auto& program = input_data_poly.template as<CompositeInstruction>();
136 TaskComposerGraph task_graph;
137
138 tesseract_common::ManipulatorInfo program_manip_info =
139 program.getManipulatorInfo().getCombined(input.problem.manip_info);
140
141 auto start_task = std::make_unique<StartTask>();
142 auto start_uuid = task_graph.addNode(std::move(start_task));
143
144 std::vector<std::pair<boost::uuids::uuid, std::string>> raster_tasks;
145 raster_tasks.reserve(program.size());
146
147 // Generate all of the raster tasks. They don't depend on anything
148 std::size_t raster_idx = 0;
149 for (std::size_t idx = 1; idx < program.size() - 1; idx += 2)
150 {
151 // Get Raster program
152 auto raster_input = program[idx].template as<CompositeInstruction>();
153
154 // Set the manipulator info
155 raster_input.setManipulatorInfo(raster_input.getManipulatorInfo().getCombined(program_manip_info));
156
157 // Get Start Plan Instruction
158 const InstructionPoly& pre_input_instruction = program[idx - 1];
159 assert(pre_input_instruction.isCompositeInstruction());
160 const auto& tci = pre_input_instruction.as<CompositeInstruction>();
161 const auto* li = tci.getLastMoveInstruction();
162 assert(li != nullptr);
163 raster_input.insertMoveInstruction(raster_input.begin(), *li);
164
165 auto raster_pipeline_task = std::make_unique<RasterTaskType>("Raster #" + std::to_string(raster_idx + 1) + ": " +
166 raster_input.getDescription());
167 std::string raster_pipeline_key = raster_pipeline_task->getUUIDString();
168 auto raster_pipeline_uuid = task_graph.addNode(std::move(raster_pipeline_task));
169 raster_tasks.emplace_back(raster_pipeline_uuid, raster_pipeline_key);
170 input.data_storage.setData(raster_pipeline_key, raster_input);
171
172 task_graph.addEdges(start_uuid, { raster_pipeline_uuid });
173
174 raster_idx++;
175 }
176
177 // Loop over all transitions
178 std::vector<std::string> transition_keys;
179 transition_keys.reserve(program.size());
180 std::size_t transition_idx = 0;
181 for (std::size_t idx = 2; idx < program.size() - 2; idx += 2)
182 {
183 // Get transition program
184 auto transition_input = program[idx].template as<CompositeInstruction>();
185
186 // Set the manipulator info
187 transition_input.setManipulatorInfo(transition_input.getManipulatorInfo().getCombined(program_manip_info));
188
189 // Get Start Plan Instruction
190 const InstructionPoly& pre_input_instruction = program[idx - 1];
191 assert(pre_input_instruction.isCompositeInstruction());
192 const auto& tci = pre_input_instruction.as<CompositeInstruction>();
193 const auto* li = tci.getLastMoveInstruction();
194 assert(li != nullptr);
195 transition_input.insertMoveInstruction(transition_input.begin(), *li);
196
197 auto transition_pipeline_task = std::make_unique<TransitionTaskType>(
198 "Transition #" + std::to_string(transition_idx + 1) + ": " + transition_input.getDescription());
199 std::string transition_pipeline_key = transition_pipeline_task->getUUIDString();
200 auto transition_pipeline_uuid = task_graph.addNode(std::move(transition_pipeline_task));
201 transition_keys.push_back(transition_pipeline_key);
202
203 const auto& prev = raster_tasks[transition_idx];
204 const auto& next = raster_tasks[transition_idx + 1];
205 auto transition_mux_task =
206 std::make_unique<UpdateStartAndEndStateTask>(prev.second, next.second, transition_pipeline_key, false);
207 std::string transition_mux_key = transition_mux_task->getUUIDString();
208 auto transition_mux_uuid = task_graph.addNode(std::move(transition_mux_task));
209
210 input.data_storage.setData(transition_mux_key, transition_input);
211
212 task_graph.addEdges(transition_mux_uuid, { transition_pipeline_uuid });
213 task_graph.addEdges(prev.first, { transition_mux_uuid });
214 task_graph.addEdges(next.first, { transition_mux_uuid });
215
216 transition_idx++;
217 }
218
219 // Plan from_start - preceded by the first raster
220 auto from_start_input = program[0].template as<CompositeInstruction>();
221 from_start_input.setManipulatorInfo(from_start_input.getManipulatorInfo().getCombined(program_manip_info));
222
223 auto from_start_pipeline_task =
224 std::make_unique<FreespaceTaskType>("From Start: " + from_start_input.getDescription());
225 std::string from_start_pipeline_key = from_start_pipeline_task->getUUIDString();
226 auto from_start_pipeline_uuid = task_graph.addNode(std::move(from_start_pipeline_task));
227
228 auto update_end_state_task =
229 std::make_unique<UpdateEndStateTask>(raster_tasks[0].second, from_start_pipeline_key, false);
230 std::string update_end_state_key = update_end_state_task->getUUIDString();
231 auto update_end_state_uuid = task_graph.addNode(std::move(update_end_state_task));
232
233 input.data_storage.setData(update_end_state_key, from_start_input);
234
235 task_graph.addEdges(update_end_state_uuid, { from_start_pipeline_uuid });
236 task_graph.addEdges(raster_tasks[0].first, { update_end_state_uuid });
237
238 // Plan to_end - preceded by the last raster
239 auto to_end_input = program.back().template as<CompositeInstruction>();
240
241 to_end_input.setManipulatorInfo(to_end_input.getManipulatorInfo().getCombined(program_manip_info));
242
243 // Get Start Plan Instruction
244 const InstructionPoly& pre_input_instruction = program[program.size() - 2];
245 assert(pre_input_instruction.isCompositeInstruction());
246 const auto& tci = pre_input_instruction.as<CompositeInstruction>();
247 const auto* li = tci.getLastMoveInstruction();
248 assert(li != nullptr);
249 to_end_input.insertMoveInstruction(to_end_input.begin(), *li);
250
251 auto to_end_pipeline_task = std::make_unique<FreespaceTaskType>("To End: " + to_end_input.getDescription());
252 std::string to_end_pipeline_key = to_end_pipeline_task->getUUIDString();
253 auto to_end_pipeline_uuid = task_graph.addNode(std::move(to_end_pipeline_task));
254
255 auto update_start_state_task =
256 std::make_unique<UpdateStartStateTask>(raster_tasks.back().second, to_end_pipeline_key, false);
257 std::string update_start_state_key = update_start_state_task->getUUIDString();
258 auto update_start_state_uuid = task_graph.addNode(std::move(update_start_state_task));
259
260 input.data_storage.setData(update_start_state_key, to_end_input);
261
262 task_graph.addEdges(update_start_state_uuid, { to_end_pipeline_uuid });
263 task_graph.addEdges(raster_tasks.back().first, { update_start_state_uuid });
264
265 TaskComposerFuture::UPtr future = executor.value().get().run(task_graph, input);
266 future->wait();
267
268 if (input.isAborted())
269 {
270 info->message = "Raster subgraph failed";
271 info->elapsed_time = timer.elapsedSeconds();
272 CONSOLE_BRIDGE_logError("%s", info->message.c_str());
273 return info;
274 }
275
276 program.clear();
277 program.emplace_back(input.data_storage.getData(from_start_pipeline_key).as<CompositeInstruction>());
278 for (std::size_t i = 0; i < raster_tasks.size(); ++i)
279 {
280 CompositeInstruction segment = input.data_storage.getData(raster_tasks[i].second).as<CompositeInstruction>();
281 segment.erase(segment.begin());
282 program.emplace_back(segment);
283
284 if (i < raster_tasks.size() - 1)
285 {
286 CompositeInstruction transition = input.data_storage.getData(transition_keys[i]).as<CompositeInstruction>();
287 transition.erase(transition.begin());
288 program.emplace_back(transition);
289 }
290 }
291 CompositeInstruction to_end = input.data_storage.getData(to_end_pipeline_key).as<CompositeInstruction>();
292 to_end.erase(to_end.begin());
293 program.emplace_back(to_end);
294
295 input.data_storage.setData(output_keys_[0], program);
296
297 info->message = "Successful";
298 info->return_value = 1;
299 info->elapsed_time = timer.elapsedSeconds();
300 return info;
301 }

303 static void checkTaskInput(const tesseract_common::AnyPoly& input)
304 {
305 // -------------
306 // Check Input
307 // -------------
308 if (input.isNull())
309 throw std::runtime_error("RasterMotionTask, input is null");
310
311 if (input.getType() != std::type_index(typeid(CompositeInstruction)))
312 throw std::runtime_error("RasterMotionTask, input is not a composite instruction");
313
314 const auto& composite = input.as<CompositeInstruction>();
315
316 // Check from_start
317 if (!composite.at(0).isCompositeInstruction())
318 throw std::runtime_error("RasterMotionTask, from_start should be a composite");
319
320 // Check rasters and transitions
321 for (std::size_t index = 1; index < composite.size() - 1; index++)
322 {
323 // Both rasters and transitions should be a composite
324 if (!composite.at(index).isCompositeInstruction())
325 throw std::runtime_error("RasterMotionTask, Both rasters and transitions should be a composite");
326 }
327
328 // Check to_end
329 if (!composite.back().isCompositeInstruction())
330 throw std::runtime_error("RasterMotionTask, to_end should be a composite");
331 }
